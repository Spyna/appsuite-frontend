#!/bin/env ruby

require 'find'

Find.find(".") do
  |path|
  if (path =~ /\.js$/)
    text = File.read(path)
    File.open(path, 'w') do
      |f|
      f.puts "// NOJSHINT NODEFINECHECK"
      f.puts text
    end
  end
end
